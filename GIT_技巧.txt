

# 查看历史提交记录

# git pull 强制覆盖本地文件
$ git fetch --all
$ git reset --hard origin/master 
$ git pull

$ git tag -a v1.4 -m "my version 1.4"

# 删除本地分支
git branch -D [branchname]

# 删除远程分支
git push origin --delete [branchname]

git stash save "msg"

# 查看stsh 列表
git stash list 

# 使用 stash 内容
git stash apply stash{0}
