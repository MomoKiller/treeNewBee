# 节流防抖函数

# 防抖 
防抖函数，事件被触发后n秒后执行。如果n秒内再次触发，最后只执行最后一次
const debounce = (fn, delay) => {
	let timer = null;
	return (...args) => {
		clearTimeout(timer);
		timer = setTimeout(()=> {
			fn.apply(this, args);
		}, delay);
	}
}




# 节流函数
const throttle = (fn, delay=500) => {
	let flag = true;
	return (...args) => {
		if(!flag) return;
		flag = false;
		setTimeout(()=> {
			fn.apply(this, args);
			flag = true;
		}, delay);
	}
}
